521{

523    $normalizedTitle = trim(preg_replace('/\s+/', ' ', $title));
524    $normalizedTitle = mb_strtolower($normalizedTitle, 'UTF-8');
525    
526    // Supprimer la ponctuation optionnelle pour plus de flexibilité
527    $normalizedTitle = preg_replace('/[^\p{L}\p{N}\s]/u', '', $normalizedTitle);
528    
529    // Trier les IDs pour ordre cohérent
530    sort($userIds);
531    
532    // Créer une chaîne unique

534    
535    return hash('sha256', $uniqueString);
